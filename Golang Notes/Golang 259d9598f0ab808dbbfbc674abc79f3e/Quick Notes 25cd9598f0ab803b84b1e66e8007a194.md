```go

### ðŸ”¹ Examples

    fmt.Println("Forever")
    break
}

for i, v := range nums {
    fmt.Println(i, v)
}
```

## âš¡ Go Functions (Quick Look)

### ðŸ”¹ Syntax

```go
func name(params) returnType { ... }
func name(params) (ret1, ret2 type) { ... }
```

### ðŸ”¹ Examples

```go
// Simple
func add(a int, b int) int {
    return a + b
}

// Multiple return
func divide(a, b int) (int, error) {
    if b == 0 {
        return 0, fmt.Errorf("divide by zero")
    }
    return a / b, nil
}

// Named return
func square(n int) (result int) {
    result = n * n
    return
}
```

## âš¡ Go Interfaces (Quick Look)

### ðŸ”¹ Syntax

```go
type InterfaceName interface {
    MethodName(params) returnType
}
```

### ðŸ”¹ Examples

```go
type Shape interface {
    Area() float64
}

type Circle struct { Radius float64 }

func (c Circle) Area() float64 {
    return 3.14 * c.Radius * c.Radius
}

var s Shape = Circle{Radius: 5}
fmt.Println(s.Area())
```

## âš¡ Go Channels (Quick Look)

### ðŸ”¹ Syntax

```go
ch := make(chan type)
ch <- value     // send
v := <-ch       // receive
```

### ðŸ”¹ Examples

```go
ch := make(chan int)

// Goroutine send
go func() { ch <- 42 }()

// Receive
val := <-ch
fmt.Println(val)

// Buffered channel
ch2 := make(chan string, 2)
ch2 <- "Hi"
ch2 <- "Go"
fmt.Println(<-ch2, <-ch2)
```
